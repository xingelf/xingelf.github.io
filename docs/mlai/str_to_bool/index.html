Converts boolean characters to boolean type if provided as ’t’ or ‘f’ in String/Object type columns.

def str_to_bool(s):
    if s == 't':
        return True
    else:
        return False

Usage

col = "Availability"
df[col] = df[col].apply(str_to_bool)
